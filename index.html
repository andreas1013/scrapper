function generateHtaccess() {
  const urls = Object.keys(pages).sort();
  let out = '# 301 Redirects — Generated by SiteCrawler\n';
  out += 'RewriteEngine On\n\n';
  out += '# Remove trailing slashes\n';
  out += 'RewriteCond %{REQUEST_FILENAME} !-d\n';
  out += 'RewriteRule ^(.*)/$ /$1 [R=301,L]\n\n';
  out += '# Page redirects\n';
  urls.forEach(url => {
    try {
      const slug = new URL(url).pathname;
      if (slug && slug !== '/') {
        out += `Redirect 301 ${slug} /${urlToPath(url).replace('/index.html','')}\n`;
      }
    } catch {}
  });
  return out;
}

async function exportStaticSite() {
  // Check JSZip loaded
  if (typeof JSZip === 'undefined') {
    alert('JSZip library failed to load. Check your internet connection and try refreshing the page.');
    return;
  }

  const total = Object.keys(pages).length;
  if (total === 0) { alert('No pages crawled yet. Run a crawl first.'); return; }

  const sitename = document.getElementById('exp-sitename').value || 'My Site';
  const primaryColor = document.getElementById('exp-color').value || '#2c7a6e';
  const navstyle = document.getElementById('exp-navstyle').value;

  const btn = document.getElementById('export-btn');
  const prog = document.getElementById('export-progress');
  btn.disabled = true;
  btn.textContent = 'Building ZIP…';

  try {

  const zip = new JSZip();
  const assets = zip.folder('assets');

  // CSS
  assets.file('style.css', generateCSS(primaryColor, navstyle));

  // .htaccess
  zip.file('.htaccess', generateHtaccess());

  // Generate each page
  let count = 0;
  for (const [url, info] of Object.entries(pages)) {
    count++;
    prog.textContent = `Generating ${count}/${total}: ${info.title || url}`;
    const html = generatePageHTML(url, info, sitename, primaryColor, navstyle);
    const filePath = urlToPath(url);
    zip.file(filePath, html);
  }

  // Migration reference map (simple HTML)
  let mapRows = Object.entries(pages).map(([url, info]) => `${info.title}${url}${new URL(url).pathname}`).join('');
  zip.file('migration-map.html', `Migration Map
Migration Map
Generated by SiteCrawler — ${new Date().toLocaleString()}
${mapRows}
Page TitleOld URLOld Path
`);

  prog.textContent = 'Compressing ZIP…';
  const blob = await zip.generateAsync({type:'blob', compression:'DEFLATE', compressionOptions:{level:6}});

  prog.textContent = `Done! ${total} pages → ZIP ready.`;
  btn.disabled = false;
  btn.textContent = '📦 Download Static Site ZIP';

  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = 'static-site.zip';
  a.click();

  } catch(err) {
    prog.textContent = 'Error: ' + err.message;
    btn.disabled = false;
    btn.textContent = '📦 Download Static Site ZIP';
    alert('Export failed: ' + err.message);
    console.error(err);
  }
}

async function generateOnePage(url) {
  const apiKey = document.getElementById('ai-key').value.trim();
  if (!apiKey) { alert('Enter your Anthropic API key first.'); return; }

  const info = pages[url];
  if (!info) { alert('Page data not found.'); return; }

  // Mark as generating
  aiResults[url] = { ...(aiResults[url]||{}), status: 'generating', approved: false };
  renderAIList();

  // Build a tight prompt — just content section, not full page
  const prompt = `You are rebuilding a website page as clean semantic HTML.

Page title: ${info.title}
URL path: ${new URL(url).pathname}
Meta description: ${info.desc || 'none'}

Headings structure:
${(info.headings||[]).map(h => h.level + ': ' + h.text).join('\n') || 'none'}

Original page text content:
---
${(info.cleanText||'').slice(0, 3000)}
---

Write ONLY the inner HTML content for the 
 element.
Do NOT include , , , 
, 
, 
 tags.
Use semantic HTML: h2, h3, p, ul, ol, blockquote, table as appropriate.
Keep all factual content accurate. Make it clean and readable.
Return only the HTML, no explanation, no markdown fences.`;

  let result;
  try {
    const workerUrl = document.getElementById('worker-url').value.trim().replace(/\/$/,'');
    const res = await fetch(workerUrl + '/ai', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'X-API-Key': apiKey },
      body: JSON.stringify({
        model: 'claude-haiku-4-5-20251001',
        max_tokens: 2048,
        messages: [{ role: 'user', content: prompt }]
      }),
      signal: AbortSignal.timeout(30000)
    });
    const data = await res.json();
    if (data.error) throw new Error(data.error);
    if (data.type === 'error') throw new Error(data.error?.message || 'API error');
    result = data.content?.[0]?.text || '';
    if (!result) throw new Error('Empty response from API');
  } catch(e) {
    aiResults[url] = { status: 'error', html: null, approved: false, error: e.message };
    renderAIList();
    alert('Generation failed: ' + e.message);
    return;
  }

  // Strip any accidental markdown fences
  result = result.replace(/^```html?\n?/i, '').replace(/\n?```$/i, '').trim();

  aiResults[url] = { status: 'done', html: result, approved: false };
  renderAIList();
  // Auto-open preview
  openAIPreview(url);
}

function approveCurrent() {
  if (!currentAIUrl) return;
  aiResults[currentAIUrl] = { ...(aiResults[currentAIUrl]||{}), approved: true, status: 'done' };
  renderAIList();
  closeAIModal();
}

function regenerateCurrent() {
  if (!currentAIUrl) return;
  closeAIModal();
  generateOnePage(currentAIUrl);
}

function closeAIModal(e) {
  if (e && e.target !== document.getElementById('ai-modal')) return;
  document.getElementById('ai-modal').classList.remove('open');
  currentAIUrl = null;
}

}

// Override exportStaticSite to use AI content where approved
const _origExport = exportStaticSite;
async function exportStaticSite() {
  const approvedCount = Object.values(aiResults).filter(r=>r.approved).length;
  if (approvedCount > 0) {
    // Temporarily patch pages content for approved pages
    const origContent = {};
    Object.entries(aiResults).forEach(([url, res]) => {
      if (res.approved && res.html && pages[url]) {
        origContent[url] = pages[url].contentHTML;
        pages[url].contentHTML = res.html;
      }
    });
    await _origExport();
    // Restore
    Object.entries(origContent).forEach(([url, html]) => {
      if (pages[url]) pages[url].contentHTML = html;
    });
  } else {
    await _origExport();
  }
}
